filme = {'titulo:'star wars',
         'ano':1997,                #esses elementos são chaves ou keys
         'diretor:'Geoge Lucas'
         }

1#  print(filme.values()) isso vai me retornar todos os valores do meu dicionario
                [star wars, 1997, George Lucas]

2# print(filme.keys()) isso vai me retornar as chaves do dicionario filme
                [titulo, ano, diretor]

3# print(filme.items()) isso retorna os valores e as chaves 
        [titulo, star wars, ano, 1997, diretor, George Lucas]

 ############  IMPORTANTE ############################

VALOR: 1# Values()     valores de dentro
CHAVE: 2# Keys()       o nome das chaves 
ITEN: 3#  Items()      os dois

pode usar isso no for EXEMPLO:

for k, v in filme.items():    #k para keys e #v para values
    print(f'o {k} é o {v}') 

    vai indo for por for então o primeiro for vai ficar

    o titulo é star wars        #primeiro loop
    o ano é 1977                 #segundo loop
    o diretor é George Lucas     #terceiro loop
    o  {k}    é     {v}
